    <script type="text/babel">
        const { useState, useRef, useEffect } = React;

        const Icon = ({ name, size = 24, className = "" }) => {
            const icons = {
                play: <path d="m6 3 14 9-14 9V3z" />,
                save: <React.Fragment><path d="M19 21H5a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h11l5 5v11a2 2 0 0 1-2 2z"/><polyline points="17 21 17 13 7 13 7 21"/><polyline points="7 3 7 8 15 8"/></React.Fragment>,
                check: <polyline points="20 6 9 17 4 12" />,
                trash: <React.Fragment><path d="M3 6h18"/><path d="M19 6v14c0 1-1 2-2 2H7c-1 0-2-1-2-2V6"/><path d="M8 6V4c0-1 1-2 2-2h4c1 0 2 1 2 2v2"/></React.Fragment>,
                volume: <React.Fragment><polygon points="11 5 6 9 2 9 2 15 6 15 11 19 11 5" /><path d="M15.54 8.46a5 5 0 0 1 0 7.07" /><path d="M19.07 4.93a10 10 0 0 1 0 14.14" /></React.Fragment>,
                arrowRight: <React.Fragment><path d="M5 12h14"/><polyline points="12 5 19 12 12 19"/></React.Fragment>,
                rotate: <React.Fragment><path d="M21 12a9 9 0 1 1-9-9c2.52 0 4.93 1 6.74 2.74L21 8"/><polyline points="21 3 21 8 16 8"/></React.Fragment>,
                enter: <React.Fragment><polyline points="9 10 4 15 9 20"/><path d="M20 4v7a4 4 0 0 1-4 4H4"/></React.Fragment>,
                home: <React.Fragment><path d="m3 9 9-7 9 7v11a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2z"/><polyline points="9 22 9 12 15 12 15 22" /></React.Fragment>,
                bell: <React.Fragment><path d="M6 8a6 6 0 0 1 12 0c0 7 3 9 3 9H3s3-2 3-9" /><path d="M10.3 21a1.94 1.94 0 0 0 3.4 0" /></React.Fragment>
            };
            return (
                <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}>
                    {icons[name]}
                </svg>
            );
        };

        function App() {
            const defaultText = `apple, 蘋果, fruit, 名詞, 水果店購物, An apple a day keeps the doctor away., 一天一蘋果，醫生遠離我。\nbrave, 勇敢的, fearless/courageous, 形容詞, 描述人物性格, He is a brave firefighter., 他是一位勇敢的消防員。\nabandon, 放棄, give up/quit, 動詞, 面對困難時, Never abandon your dreams., 永遠不要放棄你的夢想。`;

            const [appState, setAppState] = useState('import'); 
            const [vocabList, setVocabList] = useState([]);
            const [mistakesList, setMistakesList] = useState([]);
            const [currentIndex, setCurrentIndex] = useState(0);
            const [userInput, setUserInput] = useState('');
            const [feedbackStatus, setFeedbackStatus] = useState('idle'); 
            const [showHint, setShowHint] = useState(false); 
            const [isFlawed, setIsFlawed] = useState(false); 
            const [isRandomized, setIsRandomized] = useState(false); 
            const [isClozeMode, setIsClozeMode] = useState(false); 
            const [isSaved, setIsSaved] = useState(false);
            const [showResetModal, setShowResetModal] = useState(false);

            const [importText, setImportText] = useState(() => localStorage.getItem('vocab_app_raw_text') || defaultText);
            const [savedSession, setSavedSession] = useState(() => {
                const s = localStorage.getItem('vocab_app_session');
                try { 
                    const session = s ? JSON.parse(s) : null; 
                    if (session && session.vocabList && session.vocabList.length > 0 && session.currentIndex < session.vocabList.length) return session;
                    return null;
                } catch(e) { return null; }
            });

            const inputRef = useRef(null);

            useEffect(() => {
                if (appState === 'testing' && inputRef.current) inputRef.current.focus();
            }, [appState, currentIndex, feedbackStatus]);

            const playSFX = (type) => {
                try {
                    const ctx = new (window.AudioContext || window.webkitAudioContext)();
                    const osc = ctx.createOscillator();
                    const gain = ctx.createGain();
                    osc.connect(gain); gain.connect(ctx.destination);
                    if (type === 'correct') {
                        osc.frequency.setValueAtTime(800, ctx.currentTime);
                        osc.frequency.exponentialRampToValueAtTime(1200, ctx.currentTime + 0.1);
                    } else {
                        osc.type = 'triangle';
                        osc.frequency.setValueAtTime(300, ctx.currentTime);
                        osc.frequency.exponentialRampToValueAtTime(150, ctx.currentTime + 0.15);
                    }
                    gain.gain.setValueAtTime(0.05, ctx.currentTime);
                    osc.start(); osc.stop(ctx.currentTime + 0.15);
                } catch(e) {}
            };

            const playAudio = (word) => {
                if ('speechSynthesis' in window) {
                    window.speechSynthesis.cancel();
                    const u = new SpeechSynthesisUtterance(word);
                    u.lang = 'en-US';
                    window.speechSynthesis.speak(u);
                }
            };

            const saveSessionToStorage = (list, mistakes, idx) => {
                const session = { vocabList: list, mistakesList: mistakes, currentIndex: idx, isRandomized, isClozeMode };
                localStorage.setItem('vocab_app_session', JSON.stringify(session));
                setSavedSession(session);
            };

            const handleManualSave = () => {
                localStorage.setItem('vocab_app_raw_text', importText);
                setIsSaved(true);
                setTimeout(() => setIsSaved(false), 2000);
            };

            const handleImport = () => {
                let list = importText.split('\n').filter(l => l.trim()).map(line => {
                    const p = line.split(',').map(s => s.trim());
                    const word = p[0] || '';
                    return {
                        word, meaning: p[1] || '',
                        synonyms: p[2] ? p[2].split('/').map(s => s.trim().toLowerCase()) : [],
                        pos: p[3] || '未提供', context: p[4] || '未提供',
                        example: p[5] || `Example for "${word}".`,
                        exampleTranslation: p[6] || '',
                        useCloze: isClozeMode ? Math.random() > 0.5 : false 
                    };
                }).filter(i => i.word && i.meaning);

                if (list.length > 0) {
                    const finalWordList = isRandomized ? list.sort(() => Math.random() - 0.5) : list;
                    setVocabList(finalWordList); setMistakesList([]); setCurrentIndex(0);
                    setUserInput(''); setFeedbackStatus('idle'); setShowHint(false); setIsFlawed(false);
                    setAppState('testing');
                    saveSessionToStorage(finalWordList, [], 0);
                }
            };

            const handleSubmit = () => {
                if (!userInput.trim() || feedbackStatus === 'correct') return;
                const target = vocabList[currentIndex];
                const input = userInput.trim().toLowerCase();
                if (input === target.word.toLowerCase()) {
                    playSFX('correct'); setFeedbackStatus('correct'); playAudio(target.word);
                } else if (target.synonyms.includes(input)) {
                    setFeedbackStatus('synonym'); setUserInput('');
                } else {
                    playSFX('wrong'); setFeedbackStatus('wrong'); setIsFlawed(true);
                }
            };

            const handleNext = () => {
                const target = vocabList[currentIndex];
                let nextMistakes = mistakesList;
                if (isFlawed && !mistakesList.some(v => v.word === target.word)) {
                    nextMistakes = [...mistakesList, target];
                    setMistakesList(nextMistakes);
                }
                if (currentIndex + 1 < vocabList.length) {
                    const nextIdx = currentIndex + 1;
                    setCurrentIndex(nextIdx); setUserInput(''); setFeedbackStatus('idle'); setShowHint(false); setIsFlawed(false);
                    saveSessionToStorage(vocabList, nextMistakes, nextIdx);
                } else {
                    localStorage.removeItem('vocab_app_session');
                    setSavedSession(null);
                    setAppState('results');
                }
            };

            const handleResume = () => {
                if (savedSession) {
                    setVocabList(savedSession.vocabList);
                    setMistakesList(savedSession.mistakesList || []);
                    setCurrentIndex(savedSession.currentIndex);
                    setIsRandomized(savedSession.isRandomized || false);
                    setIsClozeMode(savedSession.isClozeMode || false);
                    setAppState('testing');
                    setSavedSession(null);
                    setUserInput(''); setFeedbackStatus('idle'); setShowHint(false); setIsFlawed(false);
                }
            };

            const handleHardReset = () => { localStorage.clear(); window.location.reload(); };

            if (appState === 'import') {
                return (
                    <div className="max-w-6xl mx-auto px-4 py-12 animate-fade-in flex flex-col items-center">
                        {showResetModal && (
                            <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50 p-6">
                                <div className="bg-white p-10 rounded-[2rem] max-w-lg w-full text-center shadow-2xl">
                                    <h3 className="text-3xl font-bold mb-4">清空紀錄？</h3>
                                    <div className="flex gap-4">
                                        <button onClick={handleHardReset} className="flex-1 bg-red-500 text-white py-4 rounded-2xl font-bold">確定</button>
                                        <button onClick={() => setShowResetModal(false)} className="flex-1 bg-gray-100 py-4 rounded-2xl font-bold">取消</button>
                                    </div>
                                </div>
                            </div>
                        )}
                        <div className="w-full flex justify-between items-center mb-12">
                            <button onClick={() => setShowResetModal(true)} className="text-gray-300 hover:text-red-400 transition-colors"><Icon name="trash" size={20} /></button>
                            {savedSession && (
                                <button onClick={handleResume} className="flex items-center gap-3 bg-indigo-50 text-indigo-600 px-6 py-3 rounded-full animate-bounce-subtle">
                                    <Icon name="bell" size={24} /><span className="text-lg font-bold">繼續進度</span>
                                </button>
                            )}
                        </div>
                        <h1 className="text-6xl font-light text-center mb-12">建立題庫</h1>
                        <div className="flex flex-col md:flex-row gap-6 mb-12 w-full">
                            <div className="flex-1 flex items-center justify-between bg-gray-50 p-8 rounded-3xl border-2">
                                <span className="text-2xl font-medium">隨機出題</span>
                                <input type="checkbox" checked={isRandomized} onChange={() => setIsRandomized(!isRandomized)} className="w-8 h-8" />
                            </div>
                            <div className="flex-1 flex items-center justify-between bg-gray-50 p-8 rounded-3xl border-2">
                                <span className="text-2xl font-medium">語境填空</span>
                                <input type="checkbox" checked={isClozeMode} onChange={() => setIsClozeMode(!isClozeMode)} className="w-8 h-8" />
                            </div>
                        </div>
                        <div className="bg-white p-8 rounded-[3rem] border shadow-sm w-full">
                            <div className="flex justify-between items-end mb-8">
                                <label className="text-2xl text-gray-400">單字清單 (CSV格式)</label>
                                <button onClick={handleManualSave} className={`px-6 py-2 rounded-xl border-2 font-bold ${isSaved ? 'text-green-600' : 'text-gray-400'}`}>{isSaved ? '已儲存' : '儲存題庫'}</button>
                            </div>
                            <textarea className="w-full h-80 p-8 bg-gray-50 border-2 rounded-3xl text-xl font-mono focus:outline-none" value={importText} onChange={e => { setImportText(e.target.value); setIsSaved(false); }} />
                            <button onClick={handleImport} className="w-full mt-10 bg-gray-900 text-white py-8 rounded-3xl text-3xl font-bold flex items-center justify-center gap-4 active:scale-95 transition-all">開始練習</button>
                        </div>
                    </div>
                );
            }

            if (appState === 'testing') {
                const vocab = vocabList[currentIndex];
                return (
                    <div className="max-w-6xl mx-auto px-4 pt-24 text-center animate-fade-in">
                        <div className="mb-12">
                            <span className="text-xl text-gray-300 font-bold tracking-widest">{currentIndex + 1} / {vocabList.length}</span>
                        </div>
                        <h2 className="text-5xl md:text-[8rem] font-light text-gray-800 mb-16">{vocab.meaning}</h2>
                        <div className="max-w-4xl mx-auto flex flex-col gap-8">
                            <input ref={inputRef} type="text" value={userInput} onChange={e => setUserInput(e.target.value)} onKeyDown={e => { if(e.key==='Enter') feedbackStatus==='correct' ? handleNext() : handleSubmit(); }} readOnly={feedbackStatus==='correct'} className={`w-full text-center text-4xl md:text-[5rem] py-8 bg-transparent border-b-4 focus:outline-none transition-all ${feedbackStatus==='correct'?'border-green-400 text-green-600':'border-gray-300 focus:border-black'}`} placeholder="..." autoComplete="off" />
                            
                            {feedbackStatus !== 'correct' ? (
                                <div className="flex flex-col gap-8">
                                    <button onClick={handleSubmit} className="bg-blue-50 text-blue-600 py-6 rounded-3xl text-2xl font-bold">送出答案</button>
                                    {!showHint && <button onClick={() => { setShowHint(true); setIsFlawed(true); }} className="text-gray-400">取得提示</button>}
                                    {showHint && (
                                        <div className="p-8 bg-gray-50 rounded-[2rem] text-2xl animate-fade-in">
                                            <p className="font-mono tracking-widest text-4xl mb-4">{vocab.word[0]}{' _'.repeat(vocab.word.length-1)}</p>
                                            <p className="text-gray-400 text-lg">[{vocab.pos}] {vocab.context}</p>
                                        </div>
                                    )}
                                </div>
                            ) : (
                                <div className="animate-fade-in">
                                    <div className="flex justify-center items-center gap-6 mb-8">
                                        <h3 className="text-5xl font-bold">{vocab.word}</h3>
                                        <button onClick={() => playAudio(vocab.word)} className="p-4 bg-gray-100 rounded-full"><Icon name="volume" size={32} /></button>
                                    </div>
                                    <p className="text-2xl text-gray-400 mb-10 italic">"{vocab.example}"</p>
                                    <button onClick={handleNext} className="w-full bg-gray-900 text-white py-8 rounded-3xl text-2xl font-bold">下一題 (Enter)</button>
                                </div>
                            )}
                        </div>
                    </div>
                );
            }

            if (appState === 'results') {
                return (
                    <div className="max-w-6xl mx-auto px-4 flex flex-col items-center justify-center min-h-screen text-center animate-fade-in">
                        <h2 className="text-6xl font-light mb-8">練習完成</h2>
                        <button onClick={() => setAppState('import')} className="bg-gray-900 text-white px-24 py-8 rounded-full text-2xl font-bold">返回首頁</button>
                    </div>
                );
            }
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
